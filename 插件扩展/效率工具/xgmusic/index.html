// return 'order'为顺序播放, 'random'为随机播放, 'loop'为循环播放 按序三者之一
function modifyPlaybackMode() {
    // 'order'为顺序播放, 'random'为随机播放, 'loop'为循环播放
    const CONFIGMODE = ['random', 'order', 'loop'];
    // 当前的mode
    const CURRENTMODE = player.plugins.music.mode;
    let result = 'order'
    CONFIGMODE.forEach(
        (mode, index, arr) => {
            if (mode === CURRENTMODE) {
                // 不是最后一项
                if (index !== arr.length - 1) {

                    result = CONFIGMODE[index + 1];
                } else {
                    result = CONFIGMODE[0];
                }
            }
        }
    )
    return result
    // player.plugins.music.mode = result;
}
